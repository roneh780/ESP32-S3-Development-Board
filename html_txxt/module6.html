6. ESP32 Advanced Tutorials - Small Project Development

6.1 Project 1: Smart Environment Monitoring Station

Project Overview

Using ESP32-S3 development board and DHT22 temperature and humidity sensor, build an intelligent monitoring station that can real-time monitor environmental temperature and humidity and upload data via WiFi.

Required Materials

  • ESP32-S3 development board
  • DHT22 temperature and humidity sensor
  • OLED display (optional)
  • Breadboard and connecting wires

Implementation Steps

  1. Connect DHT22 sensor to GPIO pin of ESP32-S3
  2. Install DHT library and WiFi library
  3. Write code to read temperature and humidity data
  4. Upload data to IoT platform via WiFi (such as ThingSpeak, Alibaba Cloud IoT, etc.)
  5. Optional: Add OLED display to show data in real-time

Code Example

#include <WiFi.h>
#include <DHT.h>

const char* ssid = "your_SSID";
const char* password = "your_PASSWORD";
#define DHTPIN 4  // DHT22 connected to GPIO4
#define DHTTYPE DHT22  // DHT 22 (AM2302)

DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(115200);
  dht.begin();
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("WiFi connected");
}

void loop() {
  float h = dht.readHumidity();
  float t = dht.readTemperature();
  
  if (isnan(h) || isnan(t)) {
    Serial.println("Failed to read from DHT sensor!");
    return;
  }
  
  Serial.print("Humidity: ");
  Serial.print(h);
  Serial.print(" %\t");
  Serial.print("Temperature: ");
  Serial.print(t);
  Serial.println(" *C");
  
  // Add code here to upload data to IoT platform
  
  delay(2000);
}

6.2 Project 2: WiFi Smart Switch

Project Overview

Utilize the WiFi function of ESP32-S3 to build a smart switch that can be remotely controlled through mobile APP or web page, used to control home appliances.

Required Materials

  • ESP32-S3 development board
  • Relay module
  • Button (optional)
  • Breadboard and connecting wires

Implementation Steps

  1. Connect relay module to GPIO pin of ESP32-S3
  2. Write code to create WiFi server
  3. Design simple web control interface
  4. Implement relay control logic
  5. Optional: Add local button control function

Code Example

#include <WiFi.h>

const char* ssid = "your_SSID";
const char* password = "your_PASSWORD";
const int relayPin = 5;  // Relay connected to GPIO5

WiFiServer server(80);

void setup() {
  pinMode(relayPin, OUTPUT);
  digitalWrite(relayPin, LOW);
  
  Serial.begin(115200);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("WiFi connected");
  Serial.print("IP address: ");
  Serial.println(WiFi.localIP());
  server.begin();
}

void loop() {
  WiFiClient client = server.available();
  if (client) {
    String request = client.readStringUntil('\r');
    client.println("HTTP/1.1 200 OK");
    client.println("Content-type:text/html");
    client.println();
    
    if (request.indexOf("/ON") != -1) {
      digitalWrite(relayPin, HIGH);
    } else if (request.indexOf("/OFF") != -1) {
      digitalWrite(relayPin, LOW);
    }
    
    client.println("<html><body>");
    client.println("<h1>ESP32 WiFi Smart Switch</h1>");
    client.println("<p><a href=\"/ON\"><button>Turn ON</button></a></p>");
    client.println("<p><a href=\"/OFF\"><button>Turn OFF</button></a></p>");
    client.println("</body></html>");
    client.println();
    client.stop();
  }
}

6.3 Project 3: Bluetooth Low Energy Device

Project Overview

Using the Bluetooth Low Energy (BLE) function of ESP32-S3, build a device that can communicate with mobile APP, such as Bluetooth remote control or sensor data transmission device.

Required Materials

  • ESP32-S3 development board
  • Mobile phone (install nRF Connect or similar BLE debugging APP)

Implementation Steps

  1. Initialize BLE service and characteristic
  2. Set up BLE advertising
  3. Implement data read/write functions
  4. Test communication with mobile APP

Code Example

#include <BLEDevice.h>
#include <BLEUtils.h>
#include <BLEServer.h>

#define SERVICE_UUID        "4fafc201-1fb5-459e-8fcc-c5c9c331914b"
#define CHARACTERISTIC_UUID "beb5483e-36e1-4688-b7f5-ea07361b26a8"

BLEServer* pServer = NULL;
BLECharacteristic* pCharacteristic = NULL;
bool deviceConnected = false;

class MyServerCallbacks: public BLEServerCallbacks {
    void onConnect(BLEServer* pServer) {
      deviceConnected = true;
    };
    void onDisconnect(BLEServer* pServer) {
      deviceConnected = false;
    }
};

void setup() {
  Serial.begin(115200);
  BLEDevice::init("ESP32 BLE Device");
  pServer = BLEDevice::createServer();
  pServer->setCallbacks(new MyServerCallbacks());
  BLEService *pService = pServer->createService(SERVICE_UUID);
  pCharacteristic = pService->createCharacteristic(
                      CHARACTERISTIC_UUID,
                      BLECharacteristic::PROPERTY_READ |
                      BLECharacteristic::PROPERTY_WRITE
                    );
  pCharacteristic->setValue("Hello BLE");
  pService->start();
  BLEAdvertising *pAdvertising = BLEDevice::getAdvertising();
  pAdvertising->addServiceUUID(SERVICE_UUID);
  pAdvertising->setScanResponse(true);
  pAdvertising->setMinPreferred(0x06);
  pAdvertising->setMinPreferred(0x12);
  BLEDevice::startAdvertising();
  Serial.println("Waiting for a client connection...");
}

void loop() {
  if (deviceConnected) {
    // You can add data transmission logic here
    delay(1000);
  }
}

6.4 Project 4: USB Device Simulation

Project Overview

Utilize the USB-OTG function of ESP32-S3 to simulate USB devices, such as USB keyboard, mouse, or storage device.

Required Materials

  • ESP32-S3 development board (supporting USB-OTG)
  • USB data cable
  • Computer or other USB host device

Implementation Steps

  1. Enable USB-OTG function of ESP32-S3 (short corresponding jumper)
  2. Install USB HID library
  3. Write code to simulate USB device
  4. Test connection with computer

Code Example

#include <USB.h>
#include <USBHID.h>

USBHID HID;
const uint8_t REPORT_ID = 0x01;

void setup() {
  Serial.begin(115200);
  HID.begin();
  while (!HID.begin()) {
    Serial.println("HID begin failed");
    delay(1000);
  }
  Serial.println("HID ready");
}

void loop() {
  // Simulate pressing 'A' key
  uint8_t report[8] = {0};
  report[2] = 0x04;  // Scan code for 'A' key
  HID.sendReport(REPORT_ID, report, sizeof(report));
  delay(100);
  
  // Release key
  memset(report, 0, sizeof(report));
  HID.sendReport(REPORT_ID, report, sizeof(report));
  delay(2000);
}

6.5 Project 5: Low Power Sensor Node

Project Overview

Design a low-power ESP32-S3 sensor node using deep sleep mode to extend battery life, suitable for long-term monitoring applications.

Required Materials

  • ESP32-S3 development board
  • Battery or power bank
  • Sensor (such as light-dependent resistor, temperature sensor, etc.)

Implementation Steps

  1. Connect sensor to ESP32-S3
  2. Write code to implement sensor data collection
  3. Configure deep sleep mode
  4. Test battery life

Code Example

#include <WiFi.h>

const char* ssid = "your_SSID";
const char* password = "your_PASSWORD";
const int sensorPin = 34;  // Sensor connected to GPIO34

void setup() {
  Serial.begin(115200);
  delay(1000);
  
  // Connect to WiFi
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("WiFi connected");
  
  // Read sensor data
  int sensorValue = analogRead(sensorPin);
  Serial.print("Sensor value: ");
  Serial.println(sensorValue);
  
  // You can add code here to upload data to server
  
  // Disconnect WiFi to save power
  WiFi.disconnect();
  
  // Configure deep sleep, wake up after 5 seconds
  esp_sleep_enable_timer_wakeup(5 * 1000000);  // 5 seconds
  esp_deep_sleep_start();
}

void loop() {
  // Will not execute here after deep sleep, will restart
}
